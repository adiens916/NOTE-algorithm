      while line_end_index < X:
          increment += 1
          line_end_index += increment

      # 증분이 짝수면, 최댓값은 첫 열에 있음.
      if increment % 2 == 0:
          # 최댓값과 원하는 값 사이의 거리 구하기
          diff = line_end_index - X
          # 해당 거리만큼 가감해서 원하는 위치 구하기
          row = (increment - 1) - diff
          col = diff
