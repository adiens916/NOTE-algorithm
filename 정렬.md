  ```python
    for idx in range(6, 0, -1):
        if count[idx] == 3:
            prize = 10000 + idx * 1000
            break
        elif count[idx] == 2:
            prize = 1000 + idx * 100
            break
        elif count[idx] == 1:
            max_idx = max(max_idx, idx)
        else:
            prize = max_idx * 100
  ```
